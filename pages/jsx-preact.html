import { h, render } from "preact" import { useSignal } from "@preact/signals" import htm from "htm" const html = htm.bind(h) function App() { const count = useSignal(0) return html` <div> <h3>Counter</h3> <p>Counter value: ${count.value}</p> <button onClick=${() => count.value--}>-1</button> <button onClick=${() => count.value++}>+1</button> </div> `; } render(html`<${App} />`, document.getElementById("app"))
